74  // Iteramos sobre los productos de ambas ciudades

76 /* \pre Las ciudades existen en la cuenca.
77  Los inventarios de ambas ciudades no están vacios.
78  \post Se efectua el intercambio de productos entre ciudades dependiendo de las necesidades de cada una.
79 */
80  if (it1->first < it2->first) {
81  ++it1; // Avanzamos en el inventario de la primera ciudad
82  }
83  else if (it1->first > it2->first) {
84  ++it2; // Avanzamos en el inventario de la otra ciudad
85  }
86  else {
87  // Calculamos la cantidad que necesita cada ciudad
88  int cant1 = it1->second.second - it1->second.first;
89  int cant2 = it2->second.second - it2->second.first;
90  if (cant1 * cant2 < 0) { // Si las necesidades son opuestas
91  int intercambio = min(abs(cant1), abs(cant2)); // Calculamos la cantidad a intercambiar
92  bool poner = false;
93  if (cant1 < cant2) {
94  it1->second.first -= intercambio; // Actualizamos la primera ciudad
95  comerciar_ajustes(it1->first, intercambio, poner, p);
